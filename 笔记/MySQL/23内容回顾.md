-- auto_increment:主键自增，没有插入主键值的时候会默认基于上一个进行自增
-- 清空表的时候不会把主键自增重置
-- alter table 表名 auto_increment=指定索引;
-- 主键约束:主键规则的约束条件(不为空且唯一)
-- 主键字段:带有主键约束的字段就是主键字段
-- 主键值:主键字段对应的值

-- 【三】外键约束
-- 1.三个名词
-- 外键约束:外键规则的约束
-- 外键字段:带有外键约束的字段
-- 外键值:外键字段对应的值就是外键值

-- 2.基本语法
-- 先创建被关联的表
create table dep(
	id int primary key auto_increment,
    name varchar
);

-- 在创建被关联的表
create table emp(
	id int primary key auto_increment,
    dep_id int,
    foreign key (dep_id) reference dep(id)
);

-- 3.表与表之间的关系
-- (1)一对一关系
-- 一个用户对于一个用户详情
-- (2)一对多关系
-- 员工表和部门表
-- 一个员工只能对应一个部门
-- 一个部门可以有多个员工
-- (3)多对多关系
-- 作者表和图书表
-- 一个作者可以写多本书
-- 一本书可以有好几个作者
-- 在创建多对多关系的时候 要借助第三张表来存储多对多关系
-- 在创建外键关系的时候要先创建被关联表

-- (4)级联更新和级联删除
-- 被关联的表的数据删除的时候会连带着关联表中的数据一起消失
-- 某个部门被砍掉，部门下的所有员工也喜提离职大礼包
create table emp(
	id int primary key auto_increment,
    dep_id int,
    -- 创建级联更新和级联删除不要加,隔开 和 foreign key 是一句话
    foreign key (dep_id) reference dep(id)
    on delete cascade
    on update cascade
);

from table_name
order by field_name;

-- 2.降序 desc
select *
from table_name
order by field_name desc;

-- 3.支持多字段排序
select *
from table_name
order by fielda, fieldb desc;
-- 先按照fielda升序排序 然后按照fieldb降序排序


-- 【四】limit - 约束获取到的数据量
-- 1.limit n
-- 只展示n条数据
select *
from table_name
limit n;

-- 2.limit n, m
-- 从索引n开始向后取m条数据,1的情况n=0
select *
from table_name
limit n, m;

-- 【六】联表查询和子查询
-- 都是查询数据的思路
-- 1.子查询
-- 将一句完整复杂的SQL语句进行拆分，用一条语句的结果作为另一条语句的条件

-- 2.联表查询
-- 将多张表的数据先拼接成一张表的数据，然后再在当前表中查询想要的数据
-- (1)笛卡尔积
-- A 和 B 集合 拼一起 乘

-- (2)拼表关键字
-- 内连接 inner join 保留共有的数据
select *
from a
inner join b on a.name=b.aname
-- 左连接 left join 保留左的全部 右表没有的部分为null
select *
from a

-- 右连接 right join 保留右的全部 左表没有的部分为null
select *
from a
right join b on a.name=b.aname
-- 全连接 union
select *
from a
left join b on a.name=b.aname
union
select *
from a
right join b on a.name=b.aname
