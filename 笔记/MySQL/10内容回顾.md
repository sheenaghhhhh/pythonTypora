# 【一】数据库的演变过程
# 1.基于内存存储数据
# 2.基于文件 txt文档 存储数据
# 3.基于 json 格式的文件存储数据
# 4.基于 TCP 的客户端和服务端存储数据

# 【二】数据库介绍
# 1.关系型数据库
# 彼此之间有联系
# 存储形式是以表格的形式存储数据
# 每个字段会限制当前字段下的数据
# 丰富的数据库操作 增删改查

# MySQL, Oracle, SQL Server, DB2, SQLite, PostgreSQL

# 2.非关系型数据库
# 彼此之间没有必然联系
# 存储形式是以键值对的形式存储数据
# 字段没有约束限制
# 基于内存存储数据，一旦服务重启，则数据消失

# Redis, MongoDB, memecached

# 【三】SQL语句的由来
# 由于TCP的诞生，每一个公司都可以开发一个自己公司内部的数据存储方案
# 操作自己公司的数据库就必须用自己公司规定好的语句

# 如别人的公司想要使用你的数据库就必须学你公司里的操作语句
# 大家就商量出一套公用的方案
# SQL语句由此但是
# SQL语句就是用来操作关系型数据库的语句的统筹统称

# 【四】NoSQL
# NoSQL语句就是用来操作非关系型数据库的语句的统筹统称

# 【五】SQL语句的分类
# 1.数据定义语言 DDL
# 创建 create table/database
# 删除 drop table/database
# 修改  alter table/database

# 2.数据操作语言 DML
# 查看数据 select
# 插入数据 insert
# 更新数据 updata
# 删除数据 delete

# 3.数据控制语言 DCL
# 事务确认 commit
# 事务回滚 rollback
# 赋予权限 grant
# 取消权限 remove

# 【六】数据库安装
# 1.下载和安装
# 官网下载解压到目标文件夹
# 配置环境变量 bin目录
# mysql.exe 是TCP的客户端
# mysqld.exe 是TCP的服务端

# 2.配置数据库
# 创建一个 data 文件夹
# 创建一个 my.ini 文件
# 配置 my.ini
"""
[mysqld]
# 设置3306端口
port=3306
# 设置mysql的安装目录 ******
basedir="D:\Program Files\MySQL"
# 设置mysql数据库的数据的存放目录，就是前面手动创建的data目录 ******
datadir="D:\Program Files\MySQL\data"
# 允许最大连接数
max_connections=200
# 允许连接失败的次数。
max_connect_errors=10
# 服务端使用的字符集默认为utf8mb4
character-set-server=utf8mb4
# 创建新表时将使用的默认存储引擎
default-storage-engine=INNODB
# 默认使用“mysql_native_password”插件认证, mysql_native_password
default_authentication_plugin=mysql_native_password
[mysql]
# 设置mysql网络通信的默认字符集
default-character-set=utf8mb4
[client]
# 设置mysql客户端连接服务端时默认使用的端口
port=3306
# 设置mysql客户端的默认字符集
default-character-set=utf8mb4
"""

# 2.针对数据表
"""
(1)查看数据库所在的名字
select database()

(2)切换数据库
user 数据库名字;

(3)创建数据表
create table 数据库表名(
    字段名 字段类型(宽度) 约束条件,
    字段名 字段类型(宽度) 约束条件,
    字段名 字段类型(宽度) 约束条件
)
注意 字段名和字段类型必填
多个字段之间用逗号，最后不要加逗号
;才算结束
关键字可大写可小写 但建议大写

(4)查看表
查看当前库下面的所有表
show tables;
查看创建表的语句
show create table 表名;
show create table 表名 \G;
查看当前表结构
describe 表名;
desc 表明;

(5)修改表
修改表的字段类型
alter table 表名 modify 字段名 字段类型(宽度) 约束条件;
alter table 表名 change 字段名 新字段名 字段类型(宽度) 约束条件;
给表重命名
alter table 表名 rename 新表名
追加字段
alter table表名add字段名字段类型(宽度)约束条件;    --默认追加在结尾
alter table表名add字段名字段类型(宽度)约束条件first;     --追加在开头
alter table表名add字段名字段类型(宽度)约束条件after字段名;     --追加在某个字段

(6)删除表
drop table 表名;
drop table if exists 表名;

(7)删除表字段
alter table 表名 drop 字段名;
"""
