## 1.2 继承分为两种

```python
# 1.单继承
# 新类继承一个母类
class Animal:
    category = "animal"

    def eat(self):
        print(f"eat")

    def speak(self):
        print(f"speak")


class Duck(Animal):
    category = "duck"
    # 当前的鸭子类属于动物类
    # 所以继承了动物类中的所有方法和属性

    def swim(self):
        print(f"swim")


duck = Duck()
duck.eat()
duck.speak()
duck.swim()
print(duck.category)
# Duck 定义了 category 就显示 Duck定义的
# 没有 就去父类 Animal 就显示 Animal定义的

# 2.多继承
# 新类继承多个母类


class TangDuck():
    # category = "tangduck"
    def play(self):
        print("play")


class SmallDuck(TangDuck, Duck):
    def eat(self):
        print(f"small eat")


small_duck = SmallDuck()
small_duck.eat()
print(small_duck.category)
small_duck.play()
# 都有的属性 会按照定义时 自己 父类1 父类2 的顺序找到

# 3.查看类的父类
print(SmallDuck.__bases__)  # (<class '__main__.TangDuck'>, <class '__main__.Duck'>)
print(SmallDuck.__base__)  # <class '__main__.TangDuck'>
print(Animal.__bases__)  # (<class 'object'>,)
print(Animal.__base__)  # <class 'object'>
```

# 3 属性查找顺序

```python
class Foo:
    def f1(self):
        print(f"Foo 的 f1")
    def f2(self):
        print(f"Foo 的 f2")
        self.f1()


class Bar(Foo):
    def f1(self):
        print(f"Bar 的 f1")


bar = Bar()
# 调用f2
bar.f2()
# Bar里找f2 没有
# 于是去母类Foo里找f2

# f2里调用了 f1 仍然是使用Bar的f1
# 因为初始化bar时使用的 Bar 传入Foo的self仍然是实例化Bar得到的对象
# 所以在父类里使用自己有的属性 还是按照正常的顺序查找
```

# 6 派生

```python
# 派生就是继承原本母类的所有属性后 衍生出自己的独有属性

class People:
    school = "hdu"

        self.name = name
        self.age = age
        self.gender = gender


class Teacher(People):
    # 方法和属性的覆盖
    # 同名 子类覆盖父类 父类原有的消失
    def __init__(self, course):
        self.course = course


teacher = Teacher(course="math")
```

## 6.2 两种方法

```python
# 方法1
class People:
    school = "hdu"

    def __init__(self, name, age, gender):
        self.name = name
        self.age = age
        self.gender = gender


class Teacher(People):
    # 方法1 使用super
    def __init__(self, name, age, gender, course):
        super().__init__(name, age, gender)
        self.course = course


teacher = Teacher(name="sheenagh", age=23, gender="female", course="math")
print(teacher.course)
print(teacher.name)



# 方法2
class People:
    school = "hdu"

    def __init__(self, name, age, gender):
        self.name = name
        self.age = age
        self.gender = gender


class Teacher(People):
    # 方法2 使用init
    # 指名道姓的使用某个具体的类 去调用类里面写好的方法 如果有什么属性就传什么属性
    # self 必传
    # self 是当前 Teacher 的对象
    def __init__(self, name, age, gender, course):
        # 直接调用 父类 中 的 __init__ 方法
        # 调用的是函数,因而需要传入self
        People.__init__(self, name, age, gender)
        self.course = course


teacher = Teacher(name="sheenagh", age=23, gender="female", course="math")
print(teacher.course)
print(teacher.name)
```

## 6.3 注意事项

```python
class People:
    school = "清华大学"

    def __init__(self, name, age, gender):
        self.name = name
        self.age = age
        self.gender = gender

    def eat(self, animal):
        print(f"{animal} eat")


class Family:
    def __init__(self, role):
        self.role = role


class Teacher(Family, People):
    def __init__(self, role, name, age, gender, course):
        # 1.使用 super 内置函数 实现父类的属性继承
        # super 超类超的是第一个继承的父类
        super().__init__(role)
        # 2.指名道姓的使用某个具体的类 去调用类里面写好的方法 如果有什么属性就传什么属性
        # self 必传
        # self 是当前 Teacher 的对象
        People.__init__(self, name, age, gender)
        self.course = course

    def eat(self, animal):
        super().eat(animal)


teacher = Teacher(role="admin", name="tin", age=18, gender="male", course="math")
print(teacher.name)
print(teacher.role)


# super超类超的是第一个继承的母类
# 所以super好用 但适合简单的类逻辑 不适合多母情况
```

# 7 组合

```python
# 组合就是大杂烩
# 在一个类里面将另一个类的对象或者方法继承就叫组合

class Course:
    def __init__(self, name, period, price):
        self.name = name
        self.period = period
        self.price = price

    def show_info(self):
        print(f"当前课程是 {self.name} 周期是 {self.period} 价格是 {self.price}")


class Date:
    def __init__(self, year, month, day):
        self.year = year
        self.month = month
        self.day = day

    def show_birth(self):
        return f"当前生日是 {self.year}年{self.month}月{self.day}日"


class People:
    school = "toronto"

    def __init__(self, name, age, gender):
        self.name = name
        self.age = age
        self.gender = gender


class Teacher:
    # 这个老师会上课 有日期 属于人类
    def __init__(self, name, age, gender, year, month, day):
        People.__init__(self, name, age, gender)
        # 生日信息  构建了一个对象
        self.birthday = Date(year, month, day)
        # 课程信息
        self.course_list = []

    def teach(self):
        print(f"当前讲师是 :>>>> {self.name} 年龄是 {self.age} 性别是 {self.gender} {self.birthday.show_birth()}")


teacher = Teacher(
    name="dya", age=40, gender="male", year=1984, month=12, day=20

teacher.teach()

# 继承： 子类继承了父类的属性和方法 派生出自己的独有属性
```

# 8 抽象类

## 8.1 什么是抽象类

与java一样，python也有抽象类的概念但是同样需要借助模块实现，抽象类是一个特殊的类，它的特殊之处在于只能被继承，不能被实例化

## 8.2 为什么要有抽象类

- 如果说类是从一堆对象中抽取相同的内容而来的，那么抽象类就是从一堆类中抽取相同的内容而来的，内容包括数据属性和函数属性。
- 比如我们有香蕉的类，有苹果的类，有桃子的类，从这些类抽取相同的内容就是水果这个抽象的类，你吃水果时，要么是吃一个具体的香蕉，要么是吃一个具体的桃子。你永远无法吃到一个叫做水果的东西。
- 从设计角度去看，如果类是从现实对象抽象而来的，那么抽象类就是基于类抽象而来的。
- 从实现角度来看，抽象类与普通类的不同之处在于：抽象类中只能有抽象方法（没有实现功能），该类不能被实例化，只能被继承，且子类必须实现抽象方法。这一点与接口有点类似，但其实是不同的，即将揭晓答案。
