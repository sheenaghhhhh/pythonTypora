class View:
    # 定义了一些允许访问的请求方式
    http_method_names = ['get', 'post', 'put', 'patch', 'delete', 'head', 'options', 'trace']

    def __init__(self, **kwargs):
        """
        Constructor. Called in the URLconf; can contain helpful extra
        keyword arguments, and other things.
        """
        # Go through keyword arguments, and either save their values to our
        # instance, or raise an error.
        for key, value in kwargs.items():
            setattr(self, key, value)

    # classonlymethod --- classmethod -- 绑定给类的方法
    # 类可以直接调用而不需要传入任何参数 自己会自动将自己传给 cls
    # cls ---> LoginView
    @classonlymethod
    def as_view(cls, **initkwargs):
        # 遍历传入的可变长关键字参数 initkwargs 没有是空的
        for key in initkwargs:
            # 遍历 字典中的每一个键
            # 判断当前传入键是否在自己的允许访问的请求方式访问内
            if key in cls.http_method_names:
                # 如果在就抛出异常
                raise TypeError(
                    'The method name %s is not accepted as a keyword argument '
                    'to %s().' % (key, cls.__name__)
                )
            # 判断当前自己的类中有某个方法
            # 如果自己没有这个方法
            if not hasattr(cls, key):
                raise TypeError("%s() received an invalid keyword %r. as_view "
                                "only accepts arguments that are already "
                                "attributes of the class." % (cls.__name__, key))

        # 【三】看 view 里面发生了什么？
        def view(request, *args, **kwargs):
            # 【1】cls ---> LoginView
            # 调用 LoginView 传入 所有的 initkwargs 可变长关键字参数
            self = cls(**initkwargs)
            # 【2】调用对象的setup方法
            # ---> LoginView 没有 ---> View
            self.setup(request, *args, **kwargs)
            # 【3】判断自己是否有 request 属性
            if not hasattr(self, 'request'):
                # 如果没有则抛出异常
                raise AttributeError(
                    "%s instance has no 'request' attribute. Did you override "
                    "setup() and forget to call super()?" % cls.__name__
                )
            # 【4】返回了一个 dispatch 方法
            # dispatch ---> LoginView 没有 ---> View ---> 返回值是 一个三板斧对象
            return self.dispatch(request, *args, **kwargs)

        view.view_class = cls
        view.view_initkwargs = initkwargs

        # take name and docstring from class
        update_wrapper(view, cls, updated=())

        # and possible attributes set by decorators
        # like csrf_exempt from dispatch
        update_wrapper(view, cls.dispatch, assigned=())

        # 【二】LoginView.as_view() 触发 ---> View 触发 as_view
        # as_view 返回的事 内嵌函数 view 的函数地址
        return view

    # 【2.1】走到了 View 类里面的 setup 方法
    def setup(self, request, *args, **kwargs):
        # 判断自己有 get 方法 并且没有 head 方法
        if hasattr(self, 'get') and not hasattr(self, 'head'):
            # 就将 head 方法替换成了 get 方法
            self.head = self.get
        # 把 request 对象初始化进自己的属性中
        self.request = request
        self.args = args
        self.kwargs = kwargs

    # 【4.1】View 有 dispatch 方法
    def dispatch(self, request, *args, **kwargs):
        # 先将当前的请求方式变为全小写 GET --> get
        # 判断当前的请求方式是否在自己允许的请求方式范围内

            # 获取出当前自己写的 get 函数内存地址 没有就报错
            handler = getattr(self, request.method.lower(), self.http_method_not_allowed)
        else:
            handler = self.http_method_not_allowed
        # 调用 自己写的 get 函数 ---> get 函数返回了三板斧对象
        return handler(request, *args, **kwargs)

    def http_method_not_allowed(self, request, *args, **kwargs):
        logger.warning(
            'Method Not Allowed (%s): %s', request.method, request.path,
            extra={'status_code': 405, 'request': request}
        )
        return HttpResponseNotAllowed(self._allowed_methods())

    def options(self, request, *args, **kwargs):
        """Handle responding to requests for the OPTIONS HTTP verb."""
        response = HttpResponse()
        response.headers['Allow'] = ', '.join(self._allowed_methods())
        response.headers['Content-Length'] = '0'
        return response

    def _allowed_methods(self):
        return [m.upper() for m in self.http_method_names if hasattr(self, m)]

# 借助render渲染语法

# 【二】模版层语法
# 1.{%  %}
# 用来加载系统(static静态文件系统)
# 用来做逻辑输出 (for if 等)
# 2.{{  }}
# 变量输出语法 渲染后端传入的变量

# 【三】八大基本数据类型的渲染
# 1.字符串类型 正常渲染 可以str.index查到指定索引对应的元素
# 2.列表类型 正常渲染 也可以list.index查到指定索引对应的元素
# 3.元组类型 正常渲染 也可以tuple.index查到指定索引对应的元素
# 4.字典类型 正常渲染 也可以dict.key查到指定键对应的值
# 5.集合类型 正常渲染 但不能索引取值
# 6.整数类型 正常渲染
# 7.浮点数类型 正常渲染
# 8.布尔类型 正常渲染

# 【四】函数的渲染
# 在渲染的时候如果函数有返回值 就渲染函数的返回值
# 没有返回值 默认返回None
# 渲染时后端函数自动执行
# 前端模板语法无法渲染有参数的函数 (不支持向函数内传参)

# 【五】类的渲染
# 渲染类本身 - 类() 得到类的对象
# 对象的绑定方法 - 类的绑定方法 - 静态方法 都能直接调用

# 【六】对象的渲染
# 渲染对象本身
# 对象的绑定方法 - 类的绑定方法 - 静态方法 都能直接调用

# 【七】过滤器
# 1.对后端传入的数据在前端进行额外的操作
# 2.语法
# {{ value|filter_name:param }}
# {{ 变量名|过滤器名:可选参数 }}
# 3.过滤器
# (1)计算长度 length - 返回当前数据长度
# (2)默认值 default:2 - 有值渲染自己 没值渲染默认
# (3)日期格式化 date:"Y-m-d H:i:s" - 返回指定格式的时间
# (4)切片 slice:"start:end:step" - 按指定规格切片
# (5)切取摘要 truncatechars:x - 在指定字符个数之前的内容被保留 之后的被...代替
# (6)移除指定字符 cut:str - 输出把指定内容中的str去掉之后的结果
# (7)拼接字符串 join:str - 用str拼接可迭代类型数据成字符串
# (8)加法 add:str/int - 相加 注意同类型相加 可以是字符串也可以是数字
# 4.重要的过滤器
# 前 str|safe - 把 字符串格式的html代码片段 格式化成 html代码 不然的话是输出代码字符串
# 后 marksafe(str) - 把字符串格式的html代码片段 格式化成 html代码 再给前端

# 【八】模版语法逻辑
# 1.for
"""
{% for item in 可迭代类型 %}

{% endfor %}
"""
# 2.forloop
# 循环中一个重要参数
"""
{% for item in items %}
    {{ forloop.counter }} # 渲染当前元素所在的索引位置 从 1 开始渲染
    {{ forloop.counter0 }} # 渲染当前元素所在的索引位置 从 0 开始渲染
    {{ forloop.revcounter }} # 渲染当前元素所在的索引位置 倒序 开始渲染 直到 1
    {{ forloop.revcounter0 }} # 渲染当前元素所在的索引位置 倒序 开始渲染 直到 0
    {{ forloop.first }} # 判断当前循环所在位置是不是第一个元素
    {{ forloop.last }} # 判断当前循环所在位置是不是最后一个元素
{% endfor %}
"""
# 3.for - empty
"""
{% for item in 可迭代类型 %}
    # 可迭代类型有值 走这里
{% endfor %}
{% empty %}
    # 可迭代类型没值 走这里
{% endfor %}
"""
# 4.if
"""
{% if 条件 %}

{% elif 条件 %}

{% else %}

{% endif %}
"""

# 2. model(field_name=field_value).save() 创建一条数据
# 3. all() 查看当前表中的所有数据 返回的是queryset对象
# 4. .filter(field_name=field_value) 按照指定字段过滤数据 返回的也是一个queryset对象
# 5. .filter(field_name=field_value).first() 按照指定字段过滤数据 返回的也是一个queryset对象中的第一个对象
# 6. .filter(field_name=field_value).last() 按照指定字段过滤数据 返回的也是一个queryset对象中的最后一个对象
# 7. .get(field._name=field_value) 按照指定字段过滤数据返回的也是一个具体的对象 只能获取单独一个对象 没有数据也会报错
# 8. .filter(field._name=field_value).update() 按照指定字段过滤数据对所有数据进行更新
# 9. .get(field._name=field_value) 获取到具体的对象
# obj.field_name=field_value 修改当前对象的某个具体的属性
# obj.save() 触发保存
# 10. .filter(field_name=field_value).delete() 按照指定字段过滤数据删除所有符合条件的数据
# 11. .get(field_.name=field_value).delete() 按照指定字段过滤数据获取到具体的对象 删除符合条件的对象
# 12. .filter(field_name=field_value).values(field_name) 按照指定字段过滤数据获取到过滤后的数据中的某几个具体的字段值 返回的是字典
# 13. .filter(field_name=field_value).value_List(field_name) 按照指定字段过滤数据获取到过滤后的数据中的某几个具体的字段值 返回的是元组
# 14. .valve_.List(field_name).distinct() 获取到的数据中的某几个具体的字段值 对这些字段进行去重
# 15. .order_by(field_name) 按照指定条件排序 升序
# 16. .order_by(-field_name) 按照指定条件排序 降序
# 17. .order_by(field_name).reverse() 按照指定条件排序 升序 再反转顺序
# 18. .count() 对当前获取到的数据进行计数
# 19. .exclude(field_name) 筛选后的数据中排除指定字段对应的数据
# 20. .filter(field_name=field_value).exists() 对筛选后的数据进行判断 判断当前数据是否存在
# 21. .query 查看当前ORM语句 对应的 SQL语句 只对 queryset 对象生效

# 【三】神奇的双下划线查询
# 1. field_name__gt: 当前字段大于某个值
# 2. field_name__gte: 当前字段大于等于某个值
# 3. field_name__lt: 当前字段小于某个值
# 4. field_name__lte: 当前字段小于等于某个值
# 5. field_name__ln: 或条件
# 6. field_name__range: 两个条件之间的值
# 7. field_name__contains: 模糊查询
# 8. field_name__incontains: 模糊查询 不区分大小写
# 9. field_name__startswith: 查询以某个字符开头的数据
# 10. field_name__endswith: 查询以某个字符结尾的数据
# 11. field_name__year: 查询某一年的数据
# 12. field_name__day: 查询某一天的数据
# 13. field_name__其他时间单位 ...
